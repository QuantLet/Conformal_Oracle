\begin{frame}{Time Series Foundation Models}
	\vspace{-0.2cm}
	{\small
	\begin{block}{Time Series Foundation Models (TSFMs)}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item \textbf{Chronos} (Amazon): tokenisation-based, T5 backbone, 1000 ancestral samples
			\item \textbf{TimesFM 2.5} (Google): decoder-only, native quantile head (9 deciles)
			\item \textbf{Moirai 1.1-R} (Salesforce): masked encoder, mixture distributions
			\item \textbf{Lag-Llama}: LLaMA-based, Student-$t$ distribution head, lagged features
		\end{itemize}
	\end{block}
	\vspace{1mm}
	\begin{exampleblock}{Why TSFMs for VaR?}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item \textbf{Zero-shot}: no task-specific training required
			\item Generate probabilistic forecasts $\to$ quantiles $\to$ VaR directly
			\item \textbf{Open-weight}: fully reproducible (pinned HuggingFace model IDs)
			\item Context: 512 daily returns; samples: 1{,}000; $\alpha \in \{0.01, 0.025, 0.05, 0.10\}$
		\end{itemize}
	\end{exampleblock}
	}
\end{frame}

% SLIDE 12: What Does 40% Violation Mean? (NEW)
%=============================================================================
\begin{frame}{What Does 40\% Violation Mean?}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{alertblock}{The Scale of Failure}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Expected: 1 violation per 100 days (that is what 1\% VaR means)
			\item Observed (Chronos): \textbf{40 violations per 100 days}
			\item The model is not just wrong --- it is \textbf{unusable} for risk management
		\end{itemize}
	\end{alertblock}
	\vspace{0mm}
	\begin{block}{An Analogy}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Imagine a weather forecast predicting ``no rain'' --- and it rains 40\% of the time
			\item You would not carry an umbrella based on that forecast
			\item Similarly: a bank cannot hold capital based on a 40\%-violation VaR
		\end{itemize}
	\end{block}
	\vspace{0mm}
	\begin{alertblock}{Why Does This Happen?}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Chronos tokenises returns into 4{,}096 bins --- extreme tails compressed
			\item Pre-training on general time series underweights fat-tailed financial returns
			\item The model forecasts the bulk well, but \textbf{misses precisely where risk lives}
		\end{itemize}
	\end{alertblock}
	}
\end{frame}

\begin{frame}{Why This Is Surprising}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{block}{TSFMs Are Excellent At Many Things}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Forecasting the mean: competitive with ARIMA, ETS, Prophet
			\item Capturing seasonality and trends: zero-shot, no tuning
			\item Density estimation in the bulk of the distribution
		\end{itemize}
	\end{block}
	\begin{alertblock}{But They Fail Exactly Where It Matters Most}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item The extreme tail ($\alpha = 0.01$) is where risk lives
			\item Good overall forecasting $\neq$ good tail forecasting
			\item Pre-training on general time series does not teach financial fat tails
		\end{itemize}
	\end{alertblock}
	\begin{block}{Implication}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Benchmarks on mean accuracy are \textbf{not informative} for risk
			\item Tail calibration must be evaluated and corrected separately
		\end{itemize}
	\end{block}
	}
\end{frame}

% SLIDE 14: Tokenisation Compresses Tails (Chronos) — split B1
%=============================================================================
\begin{frame}{Tokenisation Compresses Tails (Chronos)}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{alertblock}{The Mechanism}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Financial returns: $-5\%$ to $+5\%$ daily range; fat tails (kurtosis $> 3$)
			\item Chronos tokenises into 4{,}096 bins $\to$ extreme returns mapped to edge bins
			\item \textbf{Result}: tail mass is compressed, VaR systematically underestimated
			\item Both Chronos-Small and Chronos-Mini exhibit the same failure mode
		\end{itemize}
	\end{alertblock}
	\vspace{0mm}
	\begin{block}{Consequence}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item The 1\% quantile is estimated from a region where token resolution is lowest
			\item Ancestral sampling cannot recover information lost during tokenisation
			\item $\hat\pi = 0.39$--$0.42$: essentially treating the 1st percentile as a median event
		\end{itemize}
	\end{block}
	\vspace{0mm}
	\begin{alertblock}{Both Variants Affected}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Chronos-Small ($\hat\pi = 0.388$) and Chronos-Mini ($\hat\pi = 0.418$): same failure mode
			\item Model size does not help --- the tokenisation bottleneck is architectural
		\end{itemize}
	\end{alertblock}
	}
\end{frame}

%=============================================================================
% SLIDE 15: Quantile Extrapolation Problem (TimesFM) — split B2
%=============================================================================
\begin{frame}{Quantile Extrapolation Problem (TimesFM)}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{block}{The Architecture}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item TimesFM outputs 9 quantile deciles (10\%--90\%) --- not tail quantiles
			\item For $\alpha = 0.01$: must extrapolate via Student-$t$ fit to the 9 deciles
			\item The fit minimises $\sum(t_{\mathrm{ppf}}(\text{level}) - q_{\mathrm{val}})^2$ via Nelder-Mead
		\end{itemize}
	\end{block}
	\begin{alertblock}{The Risk}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Extrapolation introduces parametric bias at the extreme tail
			\item The Student-$t$ fit is driven by the bulk, not the tails
			\item \textbf{But}: $\qV$ diagnoses any systematic bias introduced
		\end{itemize}
	\end{alertblock}
	\begin{block}{Empirical Result}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item TimesFM achieves $\hat\pi = 0.014$ --- close to the 1\% target
			\item $\overline{\qV} = +0.003$: minimal correction needed despite extrapolation
		\end{itemize}
	\end{block}
	}
\end{frame}

%=============================================================================
% SLIDE 16: Pre-Training Domain Mismatch — split B3
%=============================================================================
\begin{frame}{Pre-Training Domain Mismatch}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{alertblock}{The Source of the Problem}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item TSFMs trained on millions of general time series (weather, demand, traffic)
			\item These are mostly \textbf{sub-Gaussian} --- financial fat tails are underrepresented
			\item The models learn a prior that \textbf{underweights extreme events}
		\end{itemize}
	\end{alertblock}
	\begin{exampleblock}{Empirical Confirmation}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item $\qV$ correlates with asset volatility ($\bar\rho = 0.81$ for TSFMs)
			\item Worse on volatile, fat-tailed assets (BTC, NatGas, WTI)
			\item Better on low-volatility assets (bonds, stable FX)
		\end{itemize}
	\end{exampleblock}
	\begin{alertblock}{The Contrast with Parametric Models}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item GJR-GARCH: $\mathrm{Corr}(\qV, \mathrm{vol}) = -0.79$ --- adapts tails per asset
			\item TSFMs apply a \textbf{fixed prior} learned from non-financial data
		\end{itemize}
	\end{alertblock}
	}
\end{frame}

\begin{frame}{What Goes Wrong Without Calibration}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{alertblock}{The Symptoms}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Raw VaR underestimates risk $\to$ violations pile up
			\item The model ``sees'' the bulk of the distribution, but misses the tail
			\item 5 of 9 forecasters have $> 1.5\%$ violation rate at $\alpha = 0.01$
		\end{itemize}
	\end{alertblock}
	\vspace{0mm}
	\begin{block}{The Diagnostic Gap}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Kupiec test says pass/fail --- not \textbf{how much} to correct
			\item Basel Traffic Light says Green/Yellow/Red --- not \textbf{which direction}
			\item No existing backtest tells you the \textbf{magnitude and sign} of the needed correction
		\end{itemize}
	\end{block}
	\vspace{0mm}
	\begin{alertblock}{What We Need}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item A single number that is \textbf{continuous} (not binary), \textbf{signed} (direction), and \textbf{actionable} (directly corrects VaR)
			\item This is precisely what the conformal threshold $\qV$ provides
		\end{itemize}
	\end{alertblock}
	}
\end{frame}

% SLIDE 20: What is Conformal Prediction
%=============================================================================
\begin{frame}{What is Conformal Prediction?}
	\begin{columns}[T]
		\column{0.48\textwidth}
		\begin{block}{The Setup}
		\begin{itemize}\setlength{\itemsep}{3pt}
			\item You have a \textbf{black-box model} that makes predictions
			\item You \textbf{don't trust} its uncertainty estimates
			\item You have a \textbf{calibration set}: past predictions + realised outcomes
		\end{itemize}
		\end{block}

		\begin{alertblock}{The Guarantee}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item If calibration data are exchangeable with test data:
			\begin{itemize}
				\item Corrected prediction has \textbf{guaranteed $(1-\alpha)$ coverage}
				\item \textbf{No distributional assumptions} needed
			\end{itemize}
		\end{itemize}
		\end{alertblock}

		\column{0.48\textwidth}
		\begin{exampleblock}{The Key Idea}
		\begin{enumerate}\setlength{\itemsep}{3pt}
			\item Compute a \textbf{nonconformity score} on each calibration day: ``how wrong was the model?''
			\item \textbf{Sort} the scores
			\item Take the $(1-\alpha)$ quantile $\to$ correction threshold $\hat{q}$
			\item At test time: \textbf{shift} the prediction by $\hat{q}$
		\end{enumerate}
		\end{exampleblock}

		{\scriptsize Vovk, Gammerman \& Shafer (2005); Romano, Patterson \& Cand\`es (2019)}
	\end{columns}
\end{frame}

\begin{frame}{Key Insight: We Are Not Fixing the Model}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{block}{What Conformal Correction Does}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Conformal correction does \textbf{NOT} improve the predictive distribution
			\item It does \textbf{NOT} retrain the forecaster or change its parameters
			\item It fixes \textbf{one number} --- the VaR quantile --- applied post-hoc
		\end{itemize}
	\end{block}
	\vspace{0mm}
	\begin{exampleblock}{Why This Matters}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item We are not rebuilding the forecaster --- we are \textbf{recalibrating its output}
			\item The correction is a single scalar shift: $\widetilde{\VaR}_t = -(\hat{q}_t^{\mathrm{lo}} - \qV)$
			\item Any model, any architecture, any vintage --- same wrapper, same guarantee
		\end{itemize}
	\end{exampleblock}
	\vspace{0mm}
	\begin{block}{Practical Consequence}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item No access to model internals required --- only the VaR output
			\item Computational cost: $O(n \log n)$ sort on calibration set, $O(1)$ per test day
			\item Can be applied to future models without any code changes
		\end{itemize}
	\end{block}
	}
\end{frame}

\begin{frame}{One-Sided Conformal VaR Correction}
	\vspace{-0.3cm}

	\begin{columns}[T]
		\column{0.48\textwidth}
		\begin{block}{Nonconformity Score}
		\begin{itemize}\setlength{\itemsep}{0pt}
			\item For each calibration day $t$:
			\vspace{-2mm}\[s_t^V = \hat{q}_t^{\mathrm{lo}} - r_t\]\vspace{-5mm}
			\item Large $s_t^V$: VaR was \textbf{too small}
			\item Conformal quantile:
			\vspace{-2mm}\[\qV = Q_{1-\alpha}\!\left(\{s_t^V\}_{t=1}^{n_{\mathrm{cal}}}\right)\]\vspace{-5mm}
		\end{itemize}
		\end{block}

		\begin{exampleblock}{Corrected VaR}
		\begin{itemize}\setlength{\itemsep}{0pt}
			\item At test time:
			\vspace{-2mm}\[\widetilde{\VaR}_t = -(\hat{q}_t^{\mathrm{lo}} - \qV)\]\vspace{-5mm}
			\item $\qV > 0$: VaR \textbf{increases} (more conservative)
			\item $\qV < 0$: VaR \textbf{decreases} (tighter)
		\end{itemize}
		\end{exampleblock}

		\column{0.48\textwidth}
		\begin{alertblock}{Why One-Sided?}
		\begin{itemize}\setlength{\itemsep}{0pt}
			\item VaR is a \textbf{lower tail} problem
			\item Two-sided CQR inflates $\qV$ by 13--18\%
			\item Captures upper-tail noise irrelevant to VaR
			\item One-sided: tighter, matches the problem
		\end{itemize}
		\end{alertblock}

		\begin{block}{Diagnostic Power}
		\begin{itemize}\setlength{\itemsep}{0pt}
			\item Sign $=$ direction of miscalibration
			\item Magnitude $=$ severity
			\item Goes beyond binary pass/fail
			\item Continuous ranking of all forecasters
		\end{itemize}
		\end{block}
	\end{columns}
	}
\end{frame}

% SLIDE 25: Coverage Theorem
%=============================================================================
\begin{frame}{Theorem 1: Conformal VaR Coverage}
	\vspace{-0.2cm}
	{\small
	\begin{thm}[Conformal VaR Coverage]
	Under Assumptions (A1)--(A4): stationarity, geometric $\beta$-mixing $\beta(k) \leq C\rho^k$, density regularity, weak dependence:
	\vspace{-2mm}
	\[
		\P\!\left(r_{n+1} \geq -\widetilde{\VaR}_{n+1}\right)
		\;\geq\;
		1 - \alpha - \Delta_n,
	\]
	\vspace{-4mm}

	where $\Delta_n = O\!\left(\sqrt{\tfrac{\log n}{n}} + \rho_n\right) \to 0$ as $n\to\infty$.
	\end{thm}
	}
	{\footnotesize
	\begin{columns}[T]
		\column{0.48\textwidth}
		\begin{alertblock}{Why This Theorem Matters}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item Standard conformal assumes \textbf{i.i.d.} --- violated by volatility clustering
			\item Without $\beta$-mixing: no guarantee that $\qV$ converges to the right correction
			\item The theorem ensures: \textbf{even under dependence}, coverage holds asymptotically
		\end{itemize}
		\end{alertblock}

		\column{0.48\textwidth}
		\begin{block}{Properties and Validation}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item \textbf{Distribution-free}: no Gaussian/$t$ needed
			\item GARCH Corollary: $\Delta_n = O(\sqrt{\log n/n})$
			\item $n_{\mathrm{cal}} \approx 4{,}300$: floor at 88\%; empirical: 98.9\%
			\item Valid under \textbf{realistic financial dependence} --- defensible under regulatory scrutiny
		\end{itemize}
		\end{block}
	\end{columns}
	}
\end{frame}

%=============================================================================
% SLIDE 26: Why Dependence Matters (NEW)
%=============================================================================
\begin{frame}{Why Dependence Matters}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{alertblock}{Financial Returns Are NOT i.i.d.}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Volatility clustering: large moves follow large moves (GARCH effects)
			\item Autocorrelation in squared returns: $\mathrm{Corr}(r_t^2, r_{t+1}^2) > 0$
			\item Standard conformal assumes exchangeability $\to$ breaks under dependence
		\end{itemize}
	\end{alertblock}
	\vspace{0mm}
	\begin{block}{What $\beta$-Mixing Captures}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item GARCH(1,1) and GJR-GARCH: geometrically $\beta$-mixing
			\item Stochastic volatility models: $\beta$-mixing under mild conditions
			\item Regime-switching: $\beta$-mixing if transitions are ergodic
		\end{itemize}
	\end{block}
	\vspace{0mm}
	\begin{alertblock}{Why Not Just Assume i.i.d.?}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item i.i.d.\ conformal gives exact coverage --- but the assumption is false for finance
			\item Ignoring dependence: empirical coverage can deviate from nominal in clusters
			\item $\beta$-mixing: honest about the cost of dependence ($\Delta_n$ term)
		\end{itemize}
	\end{alertblock}
	}
\end{frame}

%=============================================================================
% SLIDE 27: Interpretation of Delta_n (NEW)
%=============================================================================
\begin{frame}{Interpretation of $\Delta_n$}
	\vspace{-0.2cm}
	{\small
	\begin{block}{What $\Delta_n$ Is}
		\begin{itemize}\setlength{\itemsep}{3pt}
			\item $\Delta_n$ is \textbf{NOT} the exact coverage --- it is a \textbf{coverage floor}
			\item $\P(r_{n+1} \geq -\widetilde{\VaR}) \geq 1 - \alpha - \Delta_n$
			\item With $n_{\mathrm{cal}} \approx 4{,}300$: $\Delta_n \approx 0.11$ $\to$ guaranteed coverage $\geq$ 88\%
		\end{itemize}
	\end{block}
	\vspace{3mm}
	\begin{exampleblock}{The Gap Is Reassuring, Not Alarming}
		\begin{itemize}\setlength{\itemsep}{3pt}
			\item Theoretical floor: 88\%. Empirical coverage: 98.9\%
			\item The gap (88\% vs 99\%) reflects \textbf{worst-case distribution-free bounds}
			\item It does not reflect the method's quality --- it reflects the generality of the guarantee
			\item Tighter bounds require distributional assumptions (which we avoid)
		\end{itemize}
	\end{exampleblock}
	}
\end{frame}

\begin{frame}{$\qV$ Interpretation}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{exampleblock}{Reading the Sign and Magnitude}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item $\qV > 0$ (e.g.\ Chronos: $+0.038$): VaR \textbf{underestimates} risk $\to$ widen
			\item $\qV < 0$ (e.g.\ GJR-GARCH: $-0.007$): VaR is \textbf{over-conservative} $\to$ tighten
			\item $\qV \approx 0$ (e.g.\ TimesFM: $+0.003$): \textbf{well calibrated} $\to$ minimal shift
		\end{itemize}
	\end{exampleblock}
	\begin{block}{$\qV$ Is NOT a Test Statistic}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item It is a \textbf{correction magnitude} with physical units (return units)
			\item $\qV = +0.038$ means: VaR must be widened by 3.8 pp of return
			\item No $p$-value, no critical value --- a direct, actionable number

\begin{frame}{Conformal Calibration Is Retrospective}
	\vspace{-0.3cm}
	{\footnotesize
	\begin{block}{The Unifying Principle}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Conformal calibration is \textbf{fundamentally backward-looking}
			\item It corrects \textbf{within} regimes, not \textbf{across} regime shifts
			\item This is not a deficiency of conformal --- it is inherent to \emph{any} calibration method
		\end{itemize}
	\end{block}
	\vspace{0mm}
	\begin{alertblock}{What This Means in Practice}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item A $\qV$ estimated in calm markets will be too small for a stress regime
			\item A $\qV$ estimated during a stress regime will be too large for calm markets
			\item The 70/30 split averages across regimes --- rolling $\qV$ adapts locally
		\end{itemize}
	\end{alertblock}
	\vspace{0mm}
	\begin{block}{Mitigation}
		\begin{itemize}\setlength{\itemsep}{1pt}
			\item Rolling 250-day $\qV$: adapts within regimes (Green rises to 99\%)
			\item Stressed calibration window: include stress periods for conservative $\qV$
			\item Floor $\qV \geq 0$ for regulatory use: never allow conformal to narrow VaR
		\end{itemize}
	\end{block}
	}
\end{frame}

			\item Works without distributional assumptions or retraining
			\item Dominates four alternative recalibration baselines
		\end{itemize}
	\end{exampleblock}
	\begin{alertblock}{What It Does NOT Do}
		\begin{itemize}\setlength{\itemsep}{0pt}
			\item Does not make foundation models \textbf{better forecasters} or replace parametric models
			\item Does not protect against \textbf{unprecedented} regime shifts in real time
			\item Cannot fix a poor predictive density --- only its \textbf{quantile output}
		\end{itemize}
	\end{alertblock}
	\begin{exampleblock}{The Uncomfortable Finding}
		\begin{itemize}\setlength{\itemsep}{0pt}
			\item After correction, TSFMs perform \textbf{comparably} to GJR-GARCH --- not better
			\item Fine-tuning would reduce $|\qV|$ but likely not eliminate the need for CP
			\item TSFMs add value for \textbf{zero-shot deployment, ensembles, distributional forecasting}
		\end{itemize}
	\end{exampleblock}
	}
\end{frame}

\begin{frame}{Contributions}
	\vspace{-0.2cm}
	{\small
	\begin{block}{Method}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item \textbf{One-sided conformal VaR correction} with $\beta$-mixing coverage guarantee
			\item Post-hoc, forecaster-agnostic, distribution-free, $O(1)$ per test day
		\end{itemize}
	\end{block}
	\begin{exampleblock}{Diagnostic}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item $\qV$: continuous, signed ranking of any forecaster's tail reliability
			\item Adapts to volatility regimes; detects drift before binary backtests
		\end{itemize}
	\end{exampleblock}
	\begin{block}{Empirical Result}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item 9 models $\times$ 24 assets: \textcolor{Forest}{\textbf{187/216}} Green (static), \textcolor{Forest}{\textbf{213/216}} (rolling)
			\item Genuine recalibration: 88.7\% Green; headline not inflated by Chronos replacement
			\item Conformal dominates 4 baselines; estimation error beats flexibility at the 1\% tail
		\end{itemize}
	\end{block}
	}
\end{frame}

%=============================================================================
% SLIDE 48: Takeaways
%=============================================================================
\begin{frame}{Takeaways}
	\vspace{5mm}
	\begin{enumerate}\setlength{\itemsep}{12pt}
		\item[\Large 1.] \large $\qV$ turns calibration into a \textbf{measurable object}: continuous, signed, regime-adaptive

		\item[\Large 2.] \large One scalar fixes black-box risk forecasts \textbf{without assumptions} --- and no flexible alternative does better

		\item[\Large 3.] \large At extreme quantiles, \textbf{estimation error dominates model flexibility} --- low-dimensional corrections are optimal

		\item[\Large $\to$] \large This suggests a new direction: \textbf{separating forecasting from calibration} as two distinct layers
	\end{enumerate}

% ----- BACKUP: One-Sided vs Two-Sided -----
\begin{frame}{Score Design: One-Sided vs Two-Sided}
	\begin{columns}[T]
		\column{0.48\textwidth}
		\begin{alertblock}{Two-Sided CQR}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item $s_t = \max(\hat{q}_t^{\mathrm{lo}} - r_t,\; r_t - \hat{q}_t^{\mathrm{hi}})$
			\item Captures upper-tail noise irrelevant to VaR
			\item Inflates $\qV$ by 13--18\%
			\item Wider intervals for same Green Zone rate
		\end{itemize}
		\end{alertblock}

		\column{0.48\textwidth}
		\begin{exampleblock}{One-Sided (This Paper)}
		\begin{itemize}\setlength{\itemsep}{2pt}
			\item $s_t = \hat{q}_t^{\mathrm{lo}} - r_t$
			\item Only penalises downside misses
			\item Tighter corrections, same Green Zone
			\item VaR width 0.93--1.05$\times$ GJR-GARCH
		\end{itemize}
		\end{exampleblock}
	\end{columns}
\end{frame}

\begin{frame}{Practitioner Decision Guide}
	\vspace{-0.3cm}
	{\small
	\begin{exampleblock}{Single-Asset Regulatory VaR}
		GJR-GARCH + static conformal correction. Lowest width, indistinguishable QS from best TSFM.
	\end{exampleblock}
	\begin{block}{Zero-Shot Deployment Across Large Asset Universes}
		TSFM (TimesFM or Lag-Llama) + rolling conformal correction. No per-asset estimation required.
	\end{block}
	\begin{block}{Ensemble / Distributional Forecasting}
		Combine TSFM forecasts with conformal layer as quality filter; use $|\qV|/|\VaR|$ to screen out replacement-regime models.
	\end{block}
	\begin{alertblock}{Screening Tool}
		Use $\qV$ ranking to select base forecasters before deployment; models with $|\qV| > |\VaR|$ contribute no predictive content beyond a historical quantile.
	\end{alertblock}
	}
